AdvertisingSlot.js


import React, { Component } from 'react';
import connectToAdServer from './connectToAdServer';

class AdvertisingSlot extends Component {
  componentDidMount() {
    const { activate, id } = this.props;
    activate(id);
  }

  render() {
    return <div id={this.props.id} />
  }
}

export default connectToAdServer(AdvertisingSlot);

AdvertisingProvider.js


import React, { Component } from 'react';
import AdvertisingContext from './AdvertisingContext';

export default class extends Component {
  activate(id) {
    document.getElementById(id).innerHTML =
      '<img src="https://tpc.googlesyndication.com/' +
      'simgad/14142086296742558944" ' +
      'alt="" class="img_ad" width="300" ' +
      'height="250" border="0">';
  }

  render() {
    return (
      <AdvertisingContext.Provider value={this.activate}>
        {this.props.children}
      </AdvertisingContext.Provider>
    );
  }
}
